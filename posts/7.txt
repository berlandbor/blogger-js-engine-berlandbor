project-root/
│
├── index.html                 # Главная страница блога
│   ├── Подключает: 
│   │   ├── styles.css          # Стили сайта
│   │   ├── main/CoreBlog.js    # Основной скрипт логики блога
│   │   ├── main/linkify.js     # Обработчик ссылок и медиа-контента
│   │   └── main/speech.js      # Плагин озвучки текста
│   │
│   └── Элементы страницы:
│       ├── Оглавление (TOC)
│       ├── Поле поиска
│       ├── Блок для статей
│       └── Модальное окно для настроек озвучки
│
├── styles.css                  # Основной файл стилей
│
├── main/                       # Папка с JavaScript скриптами
│   ├── CoreBlog.js             # Основная логика блога
│   │   ├── Загружает статьи из posts/list.txt
│   │   ├── Использует linkify.js для форматирования контента
│   │   └── Интегрирует speech.js для озвучки статей
│   │
│   ├── linkify.js              # Форматирует контент постов
│   │   ├── Превращает URL в кликабельные ссылки
│   │   ├── Вставляет изображения, видео и аудио
│   │   └── Поддержка YouTube, Google Drive и HLS (.m3u8)
│   │
│   └── speech.js               # Плагин озвучки текста
│       ├── Озвучка текста статьи
│       ├── Выбор языка и голоса
│       ├── Настройка скорости и тональности
│       └── Модальное окно для управления настройками
│
└── posts/                      # Папка с файлами статей
    ├── list.txt                # Список статей (пути к *.txt)
    └── *.txt                   # Файлы с постами
        └── Формат: 
            ├── Первая строка — заголовок
            ├── Вторая строка — дата публикации
            └── Остальной текст — контент статьи
